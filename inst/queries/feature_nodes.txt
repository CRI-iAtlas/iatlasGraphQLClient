query FeatureNodes(
  $dataSet: [String!]
  $feature: [String!]
  $maxScore: Float
  $minScore: Float
  $network: [String!]
  $nTags: Int
  $related: [String!]
  $paging: PagingInput
  $distinct: Boolean
) {
  nodes(
    dataSet: $dataSet
    feature: $feature
    maxScore: $maxScore
    minScore: $minScore
    network: $network
    nTags: $nTags
    related: $related
    paging: $paging
    distinct: $distinct
  ) {
    items{
      label
      name
      score
      x
      y
      tag1 {
        characteristics
        color
        name
        longDisplay
        order
        shortDisplay
        type
      }
      tag2 {
        characteristics
        color
        name
        longDisplay
        order
        shortDisplay
        type
      }
      feature {
        name
        display
      }
    }
    paging {
      type
      pages
      total
      page
      limit
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    error
  }
}
