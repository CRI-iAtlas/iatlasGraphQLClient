query Cells(
      $paging: PagingInput
      $distinct:Boolean
      $cohort: [String!]
      $cell: [String!]
  ) {
      cells(
          paging: $paging
          distinct: $distinct
          cohort: $cohort
          cell: $cell
      ) {
        items {
            name
            type
        }
        paging {
            type
            pages
            total
            startCursor
            endCursor
            hasPreviousPage
            hasNextPage
            page
            limit
        }
        error
    }
}
